Ejemplos de Underscore.js

Colecciones (Objetos y Arrays)

each

	_.each([1, 2, 3], function(element, index) {
	  console.log(index + ': ' + element);
	});

	

	_.each({one : 1, two : 2, three : 3}, function(value, key) {
	  console.log(key + ': ' + value);
	});

map

	_.map([1, 2, 3], function(num){ return num * 3; });
	
	
	
	_.map({four : 4, five : 5, six : 6}, function(num, key){ return num * 3; });

reduce

	_.reduce([1, 2, 3], function(memo, num){ return memo + num; }, 0);

find / filter

	_.find([1, 2, 3, 4, 5, 6], function(num){ return num % 2 == 0; });
	
	
	
	_.filter([1, 2, 3, 4, 5, 6], function(num){ return num % 2 == 0; });

contains

	_.contains([1, 2, 3], 3);

pluck

	var clients = [{name : 'moe', age : 40}, {name : 'larry', age : 50}, {name : 'curly', age : 60}];
	_.pluck(clients, 'name');

max / min

	var clients = [{name : 'moe', age : 40}, {name : 'larry', age : 50}, {name : 'curly', age : 60}];
	_.max(clients, function(client){ return client.age; });
	
	
	
	var numbers = [10, 5, 100, 2, 1000];
	_.min(numbers);

Arrays

compact

	_.compact([0, 1, false, 2, '', 3]);

without

	_.without([1, 2, 1, 0, 3, 1, 4], 0, 1);

union

	_.union([1, 2, 3], [101, 2, 1, 10], [2, 1]);

intersection

	_.intersection([1, 2, 3], [101, 2, 1, 10], [2, 1]);

difference

	_.difference([1, 2, 3, 4, 5], [5, 2, 10]);

uniq

	_.uniq([1, 2, 1, 3, 1, 4]);

zip

	_.zip(['moe', 'larry', 'curly'], [30, 40, 50], [true, false, false]);

range

	_.range(10);
	
	
	
	_.range(1, 11);
	
	
	
	_.range(0, 30, 5);
	
	
	
	_.range(0, -10, -1);

Funciones

bind

	var func = function(greeting){ 
	  return greeting + ': ' + this.name;
	};
	
	var func_bind = _.bind(func, {name : 'moe'}, 'hi');
	func_bind();
	
	
	
	var func = function(greeting){ 
	  return greeting + ': ' + this.name;
	};
	
	var func_bind = _.bind(func, {name : 'moe'});
	func_bind('hola');

delay

	var func = function() {
	  return 'Retraso de 2 segundos';
	}
	_.delay(func, 2000);

keys / values

	_.keys({one : 1, two : 2, three : 3});
	
	
	
	_.values({one : 1, two : 2, three : 3});

functions

	_.functions(_);

extend

	_.extend({name : 'moe'}, {age: 50, userid : 'moe1'});

pick / omit

	_.pick({name : 'moe', age: 50, userid : 'moe1'}, 'name', 'age');
	
	
	
	_.omit({name : 'moe', age : 50, userid : 'moe1'}, 'age');

defaults

	var iceCream = {flavor : "chocolate"};
	_.defaults(iceCream, {flavor : "vanilla", sprinkles : "lots"});

is*

	Existen múltiples funciones que nos permiten realizar comprobaciones sobre los objetos:
	  - isEqual
	  - isEmpty
	  - isArray
	  - isObject
	  - isFunction
	  - ...
	
	var moe   = {name : 'moe', luckyNumbers : [13, 27, 34]};
	var clone = {name : 'moe', luckyNumbers : [13, 27, 34]};
	console.log(moe == clone);
	console.log(_.isEqual(moe, clone));

Utilidades

mixin

	_.mixin({
	  capitalize : function(string) {
	    return string.charAt(0).toUpperCase() + string.substring(1).toLowerCase();
	  }
	});
	_("fabio").capitalize();

escape / unescape

	_.escape('Curly, Larry & Moe');
	
	
	
	_.unescape('Curly, Larry &amp; Moe');

result

	var object = {cheese: 'crumpets', stuff: function(){ return 'nonsense'; }};
	_.result(object, 'cheese');
	_.result(object, 'stuff');

template

	var compiled = _.template("hello: <%= name %>");
	compiled({name : 'moe'});
